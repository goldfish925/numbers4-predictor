function predict() { const fileInput = document.getElementById('csvFile'); if (!fileInput.files.length) { alert('CSVファイルを選んでね'); return; } const reader = new FileReader(); reader.onload = function(e) { const text = e.target.result; const lines = text.trim().split('\n').map(line => line.split(',')); const rows = lines.slice(1); const parsed = []; const today = new Date(); const thirtyDaysAgo = new Date(today); thirtyDaysAgo.setDate(today.getDate() - 30); // ✅ 12行ごとのループ中に日付をチェック for (let i = 0; i < rows.length; i += 12) { if (!rows[i + 1] || !rows[i + 2]) continue; const dateStr = rows[i + 1][0].trim(); const numberStr = rows[i + 2][0].replace(/[^0-9]/g, '').padStart(4, '0'); const dateObj = new Date(dateStr.replace(/-/g, '/')); // 🎯 ここでフィルタ if (dateObj >= thirtyDaysAgo) { parsed.push({ date: dateObj, number: numberStr }); } } if (parsed.length < 2) { alert("直近30日分のデータが少なすぎます！"); return; } parsed.sort((a, b) => b.date - a.date); const usedNumbers = parsed.map(p => p.number); // 🔢 頻出数字集計 → 予測 const digitCount = Array(10).fill(0); usedNumbers.forEach(num => { num.split('').forEach(d => digitCount[parseInt(d)]++); }); const topDigits = digitCount .map((count, digit) => ({ digit, count })) .sort((a, b) => b.count - a.count) .slice(0, 4) .map(entry => entry.digit); const predicted = topDigits.sort(() => 0.5 - Math.random()).join(''); document.getElementById('result').innerText = `予測: ${predicted}`; // 📊 頻出数字の詳細 let detailText = '頻出数字:\n'; detailText += digitCount .map((count, digit) => ` ${digit}: ${count}`) .join('\n'); document.getElementById('details').innerText = detailText; // 🌀 風車差分分析の表示 const diffs = computeKurumagasaDifferences(usedNumbers); showWindmillTable(diffs); }; reader.readAsText(fileInput.files[0]); }
