function parseCSV(csvText) { const lines = csvText.trim().split('\n'); return lines.map(line => { const [date, number] = line.split(','); return { date, number }; }); } function getMostFrequentDigits(data, digitCount = 4) { const digitFrequency = Array(10).fill(0); data.forEach(entry => { entry.number.split('').forEach(digit => { digitFrequency[parseInt(digit)]++; }); }); const rankedDigits = digitFrequency .map((count, digit) => ({ digit, count })) .sort((a, b) => b.count - a.count); return rankedDigits.slice(0, digitCount).map(item => item.digit).join(''); } function predict() { const fileInput = document.getElementById('csvFile'); const mode = document.querySelector('input[name="mode"]:checked').value; if (!fileInput.files.length) { alert("CSVファイルを選んでね！"); return; } const reader = new FileReader(); reader.onload = function (e) { const csvText = e.target.result; const allData = parseCSV(csvText); let data; if (mode === 'recent') { data = allData.slice(-20); } else if (mode === 'month') { const oneMonthAgo = new Date(); oneMonthAgo.setMonth(oneMonthAgo.getMonth() - 1); data = allData.filter(entry => new Date(entry.date) >= oneMonthAgo); } const prediction = getMostFrequentDigits(data); document.getElementById('result').innerText = `予想数字: ${prediction}`; document.getElementById('details').innerText = '（頻出数字から作られた予想だよ）'; }; reader.readAsText(fileInput.files[0]); }
